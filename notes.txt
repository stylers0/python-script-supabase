# Create new table with column Schema : 

                    DROP TABLE IF EXISTS machine_events;

                    CREATE TABLE machine_events (
                        id SERIAL PRIMARY KEY,
                        timestamp TIMESTAMP NOT NULL,
                        machine_id INT NOT NULL,
                        power_status BOOLEAN NOT NULL,
                        downtime_status BOOLEAN NOT NULL,
                        duration_minutes INT,
                        machine_status_on_time INT,
                        machine_status_off_time INT,
                        downtime_status_time INT
                    );
